/** * DOXUS GMAIL & DRIVE AUTOMATION V8.8 * Integración Total con Dashboard DX-26. */ // 1. CONFIGURACIÓN DE IDENTIDAD (Debe coincidir con index.html) const PROJECT_ID = "doxus-dashboard"; const APP_ID = "doxus-cloud-tasks"; // 2. LISTA BLANCA (WHITELIST): PEGA AQUÍ LOS CORREOS AUTORIZADOS // Solo se procesarán correos de estas direcciones. const LISTA_BLANCA = [ "tu_correo_personal@gmail.com", "socio_estratega@doxus.mx", "contacto@cliente_importante.com" ]; const MAIN_FOLDER_NAME = "DOXUS - ENTRADAS GMAIL"; /** * Función principal: Escanea correos desde Enero 2026 */ function scanGmail() { // Filtro: Desde 1 de enero de 2026 y con adjuntos específicos const query = "after:2026/01/01 (filename:pdf OR filename:xlsx OR filename:docx)"; const threads = GmailApp.search(query); if (threads.length === 0) { Logger.log("No se encontraron correos nuevos para procesar."); return; } const mainFolder = getOrCreateFolder(MAIN_FOLDER_NAME); threads.forEach(thread => { const messages = thread.getMessages(); const lastMessage = messages[messages.length - 1]; // Solo procesar si el mensaje no ha sido leído (o marcar como no leído para re-escanear) if (lastMessage.isUnread()) { const fromFull = lastMessage.getFrom(); const fromEmail = fromFull.match(/<([^>]+)>/)?.[1] || fromFull; // VALIDACIÓN DE LISTA BLANCA if (LISTA_BLANCA.length > 0 && !LISTA_BLANCA.includes(fromEmail.toLowerCase().trim())) { Logger.log("Saltando remitente no autorizado: " + fromEmail); return; } const subject = lastMessage.getSubject(); const body = lastMessage.getPlainBody(); const attachments = lastMessage.getAttachments(); const dateStr = Utilities.formatDate(new Date(), "GMT-6", "yyyy-MM-dd"); // Creamos el resumen del texto del correo para el Dashboard const textSummary = body.length > 180 ? body.substring(0, 180).replace(/\n/g, " ") + "..." : body.replace(/\n/g, " "); if (attachments.length > 0) { // Carpeta física en Drive const subFolder = mainFolder.createFolder(`${dateStr} | ${subject}`); attachments.forEach(file => { const fileName = file.getName(); subFolder.createFile(file); let category = "Validar"; let priority = "Baja"; const context = (subject + " " + body).toLowerCase(); // CLASIFICACIÓN LÓGICA SOLICITADA if (context.includes("contrato")) { category = "Contrato"; priority = "Alta"; } else if (context.includes("fianza")) { category = "Revisión de Fianzas"; priority = "Media"; } else if (context.includes("validar") || context.includes("revisar")) { category = "Validar"; priority = "Alta"; } // GENERACIÓN DE FOLIO INSTITUCIONAL (DX-26-XXXX) const folio = "DX-26-" + Math.floor(1000 + Math.random() * 9000); const taskTitle = `[📁] ${fileName}`; // ESTATUS INICIAL const initialStatus = "Sin Atender"; // Sincronización con Firebase registerInCloud(folio, taskTitle, textSummary, category, priority, initialStatus); }); // Marcar como leído para evitar duplicados en el Dashboard lastMessage.markRead(); Logger.log("Expediente Sincronizado: " + subject); } } }); } function getOrCreateFolder(name) { const folders = DriveApp.getFoldersByName(name); return folders.hasNext() ? folders.next() : DriveApp.createFolder(name); } /** * Registra el documento en la base de datos de Firebase */ function registerInCloud(folio, title, summary, category, priority, status) { const url = `https://firestore.googleapis.com/v1/projects/${PROJECT_ID}/databases/(default)/documents/artifacts/${APP_ID}/public/data/pendientes`; const payload = { fields: { folio: { stringValue: folio }, title: { stringValue: title }, summary: { stringValue: summary }, category: { stringValue: category }, priority: { stringValue: priority }, status: { stringValue: status }, created: { stringValue: new Date().toISOString() } } }; const options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true }; try { const response = UrlFetchApp.fetch(url, options); if (response.getResponseCode() !== 200) { Logger.log("Error Firebase: " + response.getContentText()); } } catch (e) { Logger.log("Falla de conexión: " + e.toString()); } }
